\def\bear@draw{%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
% minimal bounding box size 
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\path (0.06,0.15) rectangle (1.8,2.13);
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% 
% switch between 2D and 3D, from
% https://chat.stackexchange.com/transcript/message/45991801#45991801
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\ifbear@threeD
	\def\bear@part@draw##1{\shade[ball color=##1]}
\else
	\def\bear@part@draw##1{\fill[##1]}
\fi
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
% Body parts
% 
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%  mitte: x=0.925
%
% Body %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\bear@part@draw{\bear@body!70!black} (0.925,0.75) ellipse (0.55 and 0.65); 
\bear@part@draw{\bear@body!50!white} (0.925,0.7) ellipse (0.35 and 0.4); 
%
% Feet %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\bear@part@draw{\bear@body!70!black} (1.35,0.3) circle (0.28); 
\bear@part@draw{\bear@body!70!black} (0.50,0.3) circle (0.28);
\bear@part@draw{\bear@body!50!white} (1.35,0.3) circle (0.17); 
\bear@part@draw{\bear@body!50!white} (0.50,0.3) circle (0.17);
%
% Ears %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\bear@part@draw{\bear@body!70!black} (1.30,1.95) circle (0.15);
\bear@part@draw{\bear@body!70!black} (0.55,1.95) circle (0.15);
\bear@part@draw{\bear@body!50!white} (1.30,1.95) circle (0.1);
\bear@part@draw{\bear@body!50!white} (0.55,1.95) circle (0.1);
%
% Head %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\bear@part@draw{\bear@body!70!black} (0.925,1.55) circle (0.5); 
%
% Muzzle %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\bear@part@draw{\bear@body!50!white} (0.925,1.4) ellipse (0.28 and 0.2); 
\bear@part@draw{\bear@body!70!white!80!red} (0.925,1.5) ellipse (0.15 and 0.08); 
%
% Eyes %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\bear@part@draw{\bear@body!30!black} (0.77,1.7) circle (0.05); 
\bear@part@draw{\bear@body!30!black} (1.08,1.7) circle (0.05); 
%
% Mouth %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\draw[\bear@body!30!black] (1.07,1.38) arc [start angle=-20, end angle=-160, radius=0.16];
%
% Arms %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\bear@part@draw{\bear@body!70!black,rotate around={-50:(1.45,0.9)}} (1.45,0.9) ellipse (0.35 and 0.15);
\bear@part@draw{\bear@body!70!black,rotate around={50:(0.4,0.9)}} (0.4,0.9) ellipse (0.35 and 0.15);
}
